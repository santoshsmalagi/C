# Pointers in C

```C

/***********************************************************************
** This example illustrates the following:                            **
** concept of pointers                                                **
** usage of address (&) and indirecton or dereferencing (*) operators **
************************************************************************/

#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[])
{
  int i = 9;
  int *ptr = &i;
  int **p = &ptr;

  // i
  printf("i = %d\n",i);
  printf("&i = %u\n",&i);
  printf("*(&i) = %u\n",*(&i));
  
  //ptr
  printf("ptr = %u\n",ptr);
  printf("&ptr = %u\n", &ptr);
  printf("*ptr = %u\n",*ptr);
  printf("*(&ptr) = %u\n",*(&ptr));
  
  //p
  printf("p = %u\n",p);
  printf("&p = %u\n", &p);
  printf("*p = %u\n",*p);
  printf("**p = %u\n",**p);
  printf("*(&p) = %u\n",*(&p));
  printf("**(&p) = %u\n",**(&p));
  return 0;
}
```
